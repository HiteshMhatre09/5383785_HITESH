#include <stdio.h>

unsigned long long largestPowerOfTwo(unsigned long long n) {
    unsigned long long p = 1;
    while (p * 2 <= n) {
        p *= 2;
    }
    return p;
}

char* counterGame(unsigned long long n) {
    int turn = 0; // 0 = Louise, 1 = Richard
    while (n > 1) {
        if ((n & (n - 1)) == 0) { 
            // n is a power of 2
            n /= 2;
        } else {
            // subtract largest power of 2 less than n
            n -= largestPowerOfTwo(n);
        }
        turn ^= 1; 
    }
    return turn ? "Louise" : "Richard";
}

int main() {
    int t;
    scanf("%d", &t);
    while (t--) {
        unsigned long long n;
        scanf("%llu", &n);
        printf("%s\n", counterGame(n));
    }
    return 0;
}
